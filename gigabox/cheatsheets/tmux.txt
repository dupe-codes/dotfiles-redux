prefix c := create new window
prefix #window_number := jump to window
prefix n/p := cycle through windows
prefix & := close window
prefix % := split panes horizontal
prefix " := split panes vertically
prefix UP/DOWN/LEFT/RIGHT := navigate panes
prefix }/{ := swap panes
prefix q := select panes with numbers
prefix z := zoom into pane
prefix ! := turn pane into window
prefix x := close pane
prefix M-left/right/up/down := resize pane
tmux new -s session_name := create new session
prefix s := list sessions
prefix w := preview open sessions
tmux attach := attach to most recent session
tmux attach -t session_name := attach to specific session
ctrl h/j/k/l := navigate around panes in vim
prefix [ := Enter copy mode with yank support
shift + alt + h/l := cycle between previous windows
prefix e := list & select open sessions
prefix f := list & select open windows
prefix :kill-session -t session_name := kill session
